export async function handleProjectCommand(
  args: string[],
  context: InteractiveContext,
  callbacks: CommandCallbacks,

    case 'info':
      await projectStatus(context);
      break;

    case 'n':

      break;

    case 'a':
      await projectAuto(args.slice(1), context, callbacks, currentConfig);
      break;

      await projectPause(context);
      break;
    case 'list':
    case 'l':
      await projectList(context);
      break;

    case 'm':
      await projectMark(args.slice(1), context);
      break;

    case 'p':
      await projectProgress(context);
      break;
